    who
        查看当前系统所有登录用户
    date
    cal
    clear
    su
    passwd
        修改密码
    man
    --help

3. 常用命令
    cd ..
    cd ~
        返回到当前用户的家目录
    cd -
        返回到上一次切换的目录
    cd
        效果等同于 cd ~

    ls -s
    ls -a
        显示所有信息
    ls -l
        显示常用信息

    pwd

    cp -i
        拷贝时进行提示(文件存在同名的情况)
    cp -r
        拷贝文件夹
    cp -p

    cp -f
        强制复制,不进行提示

    mv

    mkdir -p
        创建多级目录

    touch
        创建文件

    rm -i
    rm -r
    rm -f

    cat
        查看文件

    head -数字 文件名
        查看文件前指定行数(默认10)

    tail -数字 文件名
        查看文件后指定行数

    more
        按照百分比查看文件

    less
